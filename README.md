a=int(input("enter your first number"))
opt=int(input("enter your operator (+,-,*,/,%)"))
b=int(input("enter your second number"))
if opt=="+":
  print(a+b"ADDITION")
elif opt=="-":
    print(a-b"SUBRACTION")
elif opt=="*":
    print(a*b"MULTIPLICATION")
elif opt=="/":
    print(a/b,"division")
else:
    print("invalid operator")
